# =============================================================================
# rsync filter rules,used with --filter='. FILE' (merge directive)
# Syntax: "+ pattern" to include, "- pattern" to exclude
# Rules are evaluated top-to-bottom; first match wins.
# Parent directories must be explicitly included for rsync to traverse into them.
# =============================================================================

# =============================================================================
# General exclusions
# =============================================================================

# lock and live write-ahead log files (unreadable while applications are running)
- **/*.lock
- **/*.db-wal
- **/*.db-shm
- **/*.sqlite-wal
- **/*.sqlite-shm
- **/LOCK

# system caches
- **/.cache

# trash
- .local/share/Trash/**

# docker local data
- .local/share/docker

# backup folder itself (prevents recursive backup)
- backup

# flatpak app data (caches, reinstallable)
- .var

# development tools (reinstallable)
- .asdf

# local kubernetes / containers (ephemeral)
- .minikube
- .local/share/containers

# steam (reinstallable)
- .local/share/Steam

# claude local data
- .local/share/claude

# terraform temp files (downloaded providers, regenerable with terraform init)
- **/.terraform

# =============================================================================
# Chromium-based browsers: Brave, Chrome, Chromium
# Traversal + state + profile-dir rules are per-browser (paths differ).
# Profile content rules use ** to cover all three in one line each.
# =============================================================================

# Allow traversal (Brave has an extra BraveSoftware/Brave-Browser/ level)
+ .config/BraveSoftware/
+ .config/BraveSoftware/Brave-Browser/
+ .config/google-chrome/
+ .config/chromium/

# Browser-wide state
+ .config/BraveSoftware/Brave-Browser/Local State
+ .config/google-chrome/Local State
+ .config/chromium/Local State

# Profile directories
+ .config/BraveSoftware/Brave-Browser/Default/
+ .config/BraveSoftware/Brave-Browser/Profile */
+ .config/google-chrome/Default/
+ .config/google-chrome/Profile */
+ .config/chromium/Default/
+ .config/chromium/Profile */

# Profile content,** covers all three browsers' profiles in one rule each
+ .config/**/Bookmarks
+ .config/**/Bookmarks.bak
+ .config/**/Preferences
+ .config/**/Secure Preferences
+ .config/**/Extensions/
+ .config/**/Extensions/**
+ .config/**/Local Extension Settings/
+ .config/**/Local Extension Settings/**
+ .config/**/Extension State/
+ .config/**/Extension State/**

# Optional,uncomment to also back up history, passwords, cookies
# + .config/**/Favicons
# + .config/**/History
# + .config/**/Login Data
# + .config/**/Cookies
# + .config/**/Web Data

# Catch-all: exclude everything else in these browser directories
- .config/BraveSoftware/**
- .config/google-chrome/**
- .config/chromium/**

# =============================================================================
# Tor Browser — include profile data, exclude binary, cache, and startup cache
# Assumes torbrowser-launcher managed install on x86_64 Linux.
# For ARM or manual installs, adjust the traversal path accordingly.
# =============================================================================

# Allow traversal to the profile directory (each level required to bypass the catch-all)
+ .local/share/torbrowser/
+ .local/share/torbrowser/tbb/
+ .local/share/torbrowser/tbb/x86_64/
+ .local/share/torbrowser/tbb/x86_64/tor-browser/
+ .local/share/torbrowser/tbb/x86_64/tor-browser/Browser/
+ .local/share/torbrowser/tbb/x86_64/tor-browser/Browser/TorBrowser/
+ .local/share/torbrowser/tbb/x86_64/tor-browser/Browser/TorBrowser/Data/
+ .local/share/torbrowser/tbb/x86_64/tor-browser/Browser/TorBrowser/Data/Browser/
+ .local/share/torbrowser/tbb/x86_64/tor-browser/Browser/TorBrowser/Data/Browser/profile.default/

# Exclude cache and regenerable data inside the profile (must precede the broad include below)
- .local/share/torbrowser/**/cache2/
- .local/share/torbrowser/**/startupCache/

# Include all remaining profile data (bookmarks, prefs, extensions, certificates, logins, etc.)
+ .local/share/torbrowser/tbb/x86_64/tor-browser/Browser/TorBrowser/Data/Browser/profile.default/**

# Catch-all: exclude browser binary, libraries, and everything outside the profile
- .local/share/torbrowser/**

# =============================================================================
# VSCode,include user config, exclude cache and heavy data
# /.vscode/ is anchored (leading /) so it only matches ~/.vscode, not project
# .vscode dirs. The unanchored - .vscode catch-all below handles those.
# =============================================================================

# Allow traversal into user-data directory
+ .config/Code/
+ .config/Code/User/

# User settings and customizations
+ .config/Code/User/settings.json
+ .config/Code/User/keybindings.json
+ .config/Code/User/tasks.json
+ .config/Code/User/snippets/
+ .config/Code/User/snippets/**
+ .config/Code/User/profiles/
+ .config/Code/User/profiles/**

# Root-level .vscode dir (global startup arguments)
+ /.vscode/
+ /.vscode/argv.json

# Optional,uncomment to also back up installed extensions (can be several GBs)
# + /.vscode/extensions/
# + /.vscode/extensions/**

# Catch-all:
# - /.vscode/** excludes unincluded parts of ~/.vscode (anchored, root only)
# - .vscode excludes all project-level .vscode directories (unanchored)
# - .config/Code/** excludes logs, cache, CachedData, Workspaces, etc.
- /.vscode/**
- .vscode
- .config/Code/**
